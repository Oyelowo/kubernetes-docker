apiVersion: apps/v1
kind: Deployment
metadata:
  name: server-deployment
spec:
  replicas: 3
  selector:
    matchLabels:
      component: server
  template:
    metadata:
      labels:
        component: server
    spec:
      containers:
      - name: server-deployment
        image: stephengrider/multi-server
        resources:
          limits:
            memory: "128Mi"
            cpu: "500m"
        ports:
        - containerPort: 5000

---
apiVersion: v1
kind: Service
metadata:
  name:  server-cluster-ip-service
spec:
  type:  ClusterIP
  selector:
    component: server
  ports:
  # No nodePort here because it is not accessible from the outside world
  - name:  server-port
    port:  5000
    targetPort:  5000
